## 背景/目的

`SettingsPanel` の操作（ホットキー記録 UI・テーマ切替・autostart トグル）は
現在 E2E テストが存在しない。実装が壊れても CI で検知できない状態。

`tests/e2e/settings.spec.ts` を新規作成して主要フローを網羅する。

## 実装内容

```typescript
// tests/e2e/settings.spec.ts
import { expect, test } from '../fixtures/tauri.js';
import { waitForAppReady } from '../helpers/app-ready.js';

test.describe('設定パネル', () => {
  // 各テストで設定パネルを開くヘルパー
  async function openSettings(page) {
    await page.getByTestId('titlebar-settings-button').click();
    await expect(page.getByTestId('settings-panel')).toBeVisible();
  }

  test('設定パネルが TitleBar から開閉できること', { tag: '@smoke' }, async ({ page }) => {
    await openSettings(page);
    // 閉じる: Escape or ✕ボタン
    await page.keyboard.press('Escape');
    await expect(page.getByTestId('settings-panel')).not.toBeVisible();
  });

  test('ホットキー入力が recording 状態になること', async ({ page }) => {
    await openSettings(page);
    // 「変更」ボタンをクリックして recording 状態に
    await page.getByRole('button', { name: '変更' }).click();
    // recording 中はプレースホルダテキストが変わる
    await expect(page.locator('input[aria-label="ホットキー表示"]')).toHaveValue('キーを押してください...');
    // キャンセル
    await page.getByRole('button', { name: 'キャンセル' }).click();
  });

  test('テーマ切替が動作すること', async ({ page }) => {
    await openSettings(page);
    // 現在のテーマを確認してトグル
    const htmlEl = page.locator('html');
    const isDark = await htmlEl.evaluate((el) => el.classList.contains('dark'));
    // テーマスイッチャーをクリック
    const themeButtons = page.locator('[data-testid="theme-button"]');
    if (isDark) {
      await themeButtons.filter({ hasText: 'ライト' }).click();
      await expect(htmlEl).not.toHaveClass(/dark/);
    } else {
      await themeButtons.filter({ hasText: 'ダーク' }).click();
      await expect(htmlEl).toHaveClass(/dark/);
    }
  });
});
```

`data-testid` が未設定の箇所は合わせて付与する:

- TitleBar 設定ボタン: `data-testid="titlebar-settings-button"`（既存か確認）
- SettingsPanel 外枠: `data-testid="settings-panel"`
- テーマボタン: `data-testid="theme-button"`
